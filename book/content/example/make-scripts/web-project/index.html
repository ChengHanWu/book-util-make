Makefile

「Makefile」內容如下

THE_MAKEFILE_FILE_PATH := $(abspath $(lastword $(MAKEFILE_LIST)))
THE_BASE_DIR_PATH := $(abspath $(dir $(THE_MAKEFILE_FILE_PATH)))
THE_BIN_DIR_PATH := $(THE_BASE_DIR_PATH)/bin

.PHONY: usage web-server deploy git-add git-commit git-log git-pull git-push git-status

usage:
    @$(THE_BIN_DIR_PATH)/usage.sh

web-server:
    @$(THE_BIN_DIR_PATH)/web-server.sh

deploy:
    @$(THE_BIN_DIR_PATH)/deploy.sh

git-add:
    @$(THE_BIN_DIR_PATH)/git-add.sh

git-commit:
    @$(THE_BIN_DIR_PATH)/git-commit.sh

git-log:
        @$(THE_BIN_DIR_PATH)/git-log.sh

git-pull:
    @$(THE_BIN_DIR_PATH)/git-pull.sh

git-push:
    @$(THE_BIN_DIR_PATH)/git-push.sh

git-status:
    @$(THE_BIN_DIR_PATH)/git-status.sh

Shell Script

「bin/web-server.sh」內容如下

#!/usr/bin/env bash

THE_BASE_DIR_PATH=$(cd -P -- "$(dirname -- "$0")" && pwd -P)
THE_WEB_DIR_PATH=$THE_BASE_DIR_PATH/../web

#THE_SERVER_HOST=localhost
#THE_SERVER_HOST=127.0.0.1
THE_SERVER_HOST=0.0.0.0
THE_SERVER_PORT=8080
THE_SERVER_BIND=$THE_SERVER_HOST':'$THE_SERVER_PORT


## PHP Built-in web server
## http://php.net/manual/en/features.commandline.webserver.php

cd $THE_WEB_DIR_PATH
## php -S 0.0.0.0:8080
php -S $THE_SERVER_BIND

# php -S $THE_SERVER_BIND -t $THE_WEB_DIR_PATH


## Python
## Python 3
## https://docs.python.org/3/library/http.server.html
## python -m http.server 8080
## python -m http.server 8080 --bind 127.0.0.1

# python -m http.server $THE_SERVER_PORT
# python -m http.server $THE_SERVER_PORT --bind $THE_SERVER_HOST

## Python 2
## https://docs.python.org/2/library/simplehttpserver.html
## python -m SimpleHTTPServer 8080

# python -m SimpleHTTPServer $THE_SERVER_PORT

利用「PHP / Built-in web server」來啟動「Web Server」。

在Xubuntu環境下若要執行「php」，只要執行「sudo apt-get install php5-cli」，安裝「php5-cli」這個套件就行了。

執行

$ php -S 0.0.0.0:8080

也可以使用Pthon的版本，來啟動「Web Server」。

  • Python3 是「http.server」。
  • Python2 是「SimpleHTTPServer」。

執行

$ python3 -m http.server 8080 --bind 0.0.0.0

或是

$ python -m SimpleHTTPServer 8080

$ make
